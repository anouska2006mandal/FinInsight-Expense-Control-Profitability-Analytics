 Query 1: Display Budget Table
SELECT * FROM budget;

 Query 2: Display Expenses Table
SELECT * FROM expenses;

 Query 3: Display Revenue Table
SELECT * FROM revenue;

Query 4: Total Budget by Department
SELECT Department, SUM(Budget_Amount) AS Total_Budget
FROM budget
GROUP BY Department;

Query 5: Total Expenses by Department
SELECT Department, SUM(Amount) AS Total_Expenses
FROM expenses
GROUP BY Department;

Query 6: Total Revenue by Product
SELECT Product, SUM(Revenue) AS Total_Revenue
FROM revenue
GROUP BY Product;

Query 7: Highest Spending Department
SELECT Department, SUM(Amount) AS Total_Expenses
FROM expenses
GROUP BY Department
ORDER BY Total_Expenses DESC
LIMIT 1;

Query 8: Lowest Spending Department
SELECT Department, SUM(Amount) AS Total_Expenses
FROM expenses
GROUP BY Department
ORDER BY Total_Expenses ASC
LIMIT 1;

Query 9: Product-wise Profit
SELECT Product,
       SUM(Revenue) AS Revenue,
       SUM(Cost) AS Cost,
       SUM(Revenue-Cost) AS Profit
FROM revenue
GROUP BY Product;

Query 10:Highest Profit Product
SELECT Product,
       SUM(Revenue-Cost) AS Profit
FROM revenue
GROUP BY Product
ORDER BY Profit DESC
LIMIT 1;

Query 11: Monthly Revenue
SELECT Month,
       SUM(Revenue) AS Total_Revenue
FROM revenue
GROUP BY Month;

Query 12: Expenses Greater Than ₹10,000
SELECT *
FROM expenses
WHERE Amount > 10000;

Query 13: Revenue Greater Than ₹2,00,000
SELECT *
FROM revenue
WHERE Revenue > 200000;

Query 14: Budget in Descending Order
SELECT *
FROM budget
ORDER BY Budget_Amount DESC;

Query 15:Number of Expense Records
SELECT COUNT(*) AS Total_Expense_Records
FROM expenses;
